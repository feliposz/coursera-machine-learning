% ==========================

A(linha, coluna)  % obtem um elemento específico
A(linha, :)       % obtem uma linha inteira
A(:, coluna)      % obtem uma coluna inteira

A([linhas], :)    % obtem uma linha inteira
A(:, [colunas])   % obtem uma coluna inteira

A(:)              % pega os elementos de uma matriz bidimensional e transforma em um vetor de coluna

value = max(v) % encontra o maior elemento do vetor v
[value, index] = max(v) % encontra o maior elemento do vetor v e sua posição
% Se max for usado em uma matriz, trará o máximo de cada coluna.
max(A, [], 1) % máximo por coluna
max(A, [], 2) % máximo por linha
max(A(:)) == max(max(A))

find(A < 3) % retorna os elementos de A que são menores que 3
